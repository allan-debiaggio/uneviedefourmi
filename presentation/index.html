Algorithme 1 Chemin linéaire

Type de problématique

Une distribution des ressources étape par étape, suivant un itinéraire fixé, avec des ressources limitées.

Conditions clés :
Une seule fourmi peut occuper les salles S1 et S2

Algorithme 1 Chemin linéaire

methode_simuler.md 
# Méthode simuler()

1. On remet à zéro les compteurs d'étapes et de fourmis arrivées
2. Tant que toutes les fourmis ne sont pas arrivées :
  • On incrémente le compteur d'étapes
  • On parcourt les positions des fourmis de la dernière à la première
  • On vérifie l'occupation de la salle suivante
  • On libère la salle actuelle si nécessaire
  • On marque la salle suivante comme occupée ou la fourmi comme arrivée
  • On met à jour le compteur de fourmis arrivées
fourmiliere_1.cpp 
class Fourmi {
public:
  int id;
  int position; // index dans le chemin
  bool arrivee;
  
  Fourmi(int id_) : id(id_), position(0), arrivee(false) {}
};

class Fourmiliere1 {
public:
  std::vector<std::string> chemin;
  std::vector<std::unique_ptr<Fourmi>> fourmis;
  std::vector<int> occupation; // 0 = libre, 1 = occupée
  int nb_fourmis;

  Fourmiliere1(int n) : nb_fourmis(n) {
    chemin = {"Sv", "S1", "S2", "Sd"};
    occupation = std::vector<int>(chemin.size(), 0);
    for (int i = 0; i < n; ++i) {
      fourmis.push_back(std::make_unique<Fourmi>(i + 1));
    }
  }

  void simuler() {
    int arrived = 0;
    int step = 0;
    
    while (arrived < nb_fourmis) {
      ++step;
      // On part de la fin pour éviter les collisions
      for (int i = nb_fourmis - 1; i >= 0; --i) {
        Fourmi* f = fourmis[i].get();
        if (f->arrivee) continue;
        
        int pos = f->position;
        if (pos < (int)chemin.size() - 1) {
          // Vérifier disponibilité
          if (chemin[pos + 1] == "Sd" || occupation[pos + 1] == 0) {
            // Libérer et avancer
            if (chemin[pos] != "Sv") occupation[pos] = 0;
            f->position++;
            if (chemin[f->position] != "Sd") occupation[f->position] = 1;
            if (chemin[f->position] == "Sd") f->arrivee = true;
          }
        }
      }
      // Compter arrivées
      arrived = 0;
      for (int i = 0; i < nb_fourmis; ++i) {
        if (fourmis[i]->arrivee) arrived++;
      }
    }
  }
};

Algorithme 2 Chemins parallèles

Type de problématique

L'algorithme qui met en œuvre une approche de traitement parallèle entre deux chemins.

Caractéristiques clés :
• Répartition équilibrée des fourmis entre deux chemins
• Une seule fourmi peut occuper S1 et S2 simultanément
• Optimisation du temps total de parcours

Implémentation Algorithme 2

fourmiliere_2.cpp 
class Fourmi {
public:
  int id;
  int position; // index du chemin sur lequel elle se trouve
  int chemin;   // 0 pour Sv->S1->Sd, 1 pour Sv->S2->Sd
  bool arrivee;
  Fourmi(int id_, int chemin_) : id(id_), position(0), chemin(chemin_), arrivee(false) {}
};

class Fourmiliere2 {
public:
  std::vector<std::vector<std::string>> chemins = {
      {"Sv", "S1", "Sd"},
      {"Sv", "S2", "Sd"}};
  std::vector<std::unique_ptr<Fourmi>> fourmis;
  int occupation[2] = {0, 0}; // occupation[0] pour S1, occupation[1] pour S2
  int nb_fourmis;

  Fourmiliere2(int n) : nb_fourmis(n) {
    // Répartition équilibrée des fourmis sur les deux chemins
    for (int i = 0; i < n; ++i) {
      int ch = (i % 2 == 0) ? 0 : 1;
      fourmis.push_back(std::make_unique<Fourmi>(i + 1, ch));
    }
  }

  void simuler() {
    int arrived = 0;
    int step = 0;
    while (arrived < nb_fourmis) {
      ++step;
      // Libérer les salles S1/S2 (on le fait avant de déplacer)
      occupation[0] = 0;
      occupation[1] = 0;
      
      // On part de la fin pour éviter les collisions
      for (int i = nb_fourmis - 1; i >= 0; --i) {
        Fourmi *f = fourmis[i].get();
        if (f->arrivee)
          continue;
          
        int pos = f->position;
        int ch = f->chemin;
        std::vector<std::string> &chemin = chemins[ch];
        
        // Si pas encore à Sd
        if (pos < 2) {
          // Vérifier si la salle suivante est libre (Sd toujours libre)
          if (chemins[ch][pos + 1] == "Sd" || occupation[ch] == 0) {
            f->position++;
            if (chemins[ch][f->position] == "Sd") {
              f->arrivee = true;
            } else {
              occupation[ch] = 1; // occuper S1 ou S2
            }
          }
        }
      }
      
      // Compter les fourmis arrivées
      arrived = 0;
      for (int i = 0; i < nb_fourmis; ++i) {
        if (fourmis[i]->arrivee)
          arrived++;
      }
    }
  }
};

Algorithme 3 Gestion du comportement cyclique

Type de problématique

Algorithme mettant en œuvre un mécanisme de terminaison spécifique pour éviter les comportements cycliques.

Caractéristiques clés :
• Structure cyclique potentielle avec retour de Sd vers Sv
• Mécanisme de terminaison empêchant les fourmis de repartir après avoir atteint Sd
• Permet d'éviter les boucles infinies dans le parcours

fourmiliere_3.cpp 
class Fourmi {
public:
  int id;
  int position; // index dans le chemin
  bool arrivee;  // indicateur binaire d'arrivée
  Fourmi(int id_) : id(id_), position(0), arrivee(false) {}
};

class Fourmiliere3 {
public:
  std::vector<std::string> chemin;
  std::vector<std::unique_ptr<Fourmi>> fourmis;
  std::vector<int> occupation; // 0 = libre, 1 = occupée
  int nb_fourmis;

  Fourmiliere3(int n) : nb_fourmis(n) {
    chemin = {"Sv", "S1", "S2", "Sd"};
    occupation = std::vector<int>(chemin.size(), 0);
    for (int i = 0; i < n; ++i) {
      fourmis.push_back(std::make_unique<Fourmi>(i + 1));
    }
  }

  void simuler() {
    int arrived = 0;
    int step = 0;
    while (arrived < nb_fourmis) {
      ++step;
      // On part de la fin pour éviter les collisions
      for (int i = nb_fourmis - 1; i >= 0; --i) {
        Fourmi *f = fourmis[i].get();
        
        // Ignorer les fourmis déjà arrivées
        if (f->arrivee)
          continue;
          
        int pos = f->position;
        if (pos < (int)chemin.size() - 1) {
          // Vérifier disponibilité de la salle suivante
          if (chemin[pos + 1] == "Sd" || occupation[pos + 1] == 0) {
            // Libérer la salle actuelle (sauf Sv)
            if (chemin[pos] != "Sv")
              occupation[pos] = 0;
              
            // Avancer
            f->position++;
            
            // Occuper la nouvelle salle (sauf Sd)
            if (chemin[f->position] != "Sd")
              occupation[f->position] = 1;
              
            // Marquer définitivement l'arrivée
            if (chemin[f->position] == "Sd")
              f->arrivee = true;
          }
        }
      }
      
      // Compter les fourmis arrivées
      arrived = 0;
      for (int i = 0; i < nb_fourmis; ++i) {
        if (fourmis[i]->arrivee)
          arrived++;
      }
    }
  }
};

# Différences avec l'Algorithme 1

* Chaque fourmi possède un indicateur binaire arrivee, initialisé à false

* Lorsqu'une fourmi atteint la destination « Sd », son indicateur arrivee est définitivement mis à true

* Les fourmis marquées comme arrivées sont ignorées dans tous les traitements ultérieurs

* Même dans un graphe cyclique (avec retour de Sd vers Sv), les fourmis ne peuvent pas refaire le circuit

Algorithme 4 : Apprentissage du parcours et backtracking

Type de problématique

Algorithme simulant une navigation avec des embranchements et des impasses, en introduisant le concept d'apprentissage du chemin et de backtracking.

Caractéristiques clés :
• Détection et apprentissage des impasses (S1→S2→S3)
• Mécanisme de retour en arrière (backtracking) pour réessayer un autre chemin
• Mémorisation du bon itinéraire (S1→S4→Sd) après exploration

Implémentation Algorithme 4

fourmiliere_4.cpp 
class Fourmi {
public:
  int id;
  int position; // index dans le chemin courant
  int chemin;   // 0: Sv->S1->S2->S3 (impasse), 1: Sv->S1->S4->Sd (valide)
  bool arrivee;
  bool retour;  // true si la fourmi revient en arrière
  Fourmi(int id_, int chemin_) : id(id_), position(0), chemin(chemin_), arrivee(false), retour(false) {}
};

class Fourmiliere4 {
public:
  std::vector<std::vector<std::string>> chemins = {
      {"Sv", "S1", "S2", "S3"},
      {"Sv", "S1", "S4", "Sd"}};
  std::vector<std::unique_ptr<Fourmi>> fourmis;
  std::vector<int> occupation; // occupation des salles S1, S2, S3, S4
  int nb_fourmis;

  Fourmiliere4(int n) : nb_fourmis(n) {
    // Répartition : 1 fourmi sur le mauvais chemin, le reste sur le bon
    for (int i = 0; i < n; ++i) {
      int ch = (i == 0) ? 0 : 1;
      fourmis.push_back(std::make_unique<Fourmi>(i + 1, ch));
    }
    // occupation[0]=S1, [1]=S2, [2]=S3, [3]=S4
    occupation = std::vector<int>(4, 0);
  }

  int salle_index(const std::string &salle) {
    if (salle == "S1") return 0;
    if (salle == "S2") return 1;
    if (salle == "S3") return 2;
    if (salle == "S4") return 3;
    return -1; // Sv ou Sd
  }

  void simuler() {
    int arrived = 0;
    int step = 0;
    
    while (arrived < nb_fourmis) {
      ++step;
      // On part de la fin pour éviter les collisions
      for (int i = nb_fourmis - 1; i >= 0; --i) {
        Fourmi *f = fourmis[i].get();
        if (f->arrivee) continue;
        
        int pos = f->position;
        int ch = f->chemin;
        std::vector<std::string> &chemin = chemins[ch];
        
        // Si la fourmi est sur le mauvais chemin (impasse)
        if (ch == 0 && !f->retour) {
          if (pos < (int)chemin.size() - 1) {
            std::string next = chemin[pos + 1];
            int idx_next = salle_index(next);
            // Si la salle suivante est libre
            if (idx_next == -1 || occupation[idx_next] == 0) {
              int idx_cur = salle_index(chemin[pos]);
              if (idx_cur >= 0) occupation[idx_cur] = 0;
              f->position++;
              if (salle_index(chemin[f->position]) >= 0)
                occupation[salle_index(chemin[f->position])] = 1;
              // Si on arrive à S3, on commence le retour (impasse détectée)
              if (chemin[f->position] == "S3") {
                f->retour = true;
              }
            }
          }
        }
        else if (ch == 0 && f->retour) {
          // Retour en arrière vers Sv (backtracking)
          if (pos > 0) {
            std::string prev = chemin[pos - 1];
            int idx_prev = salle_index(prev);
            if (idx_prev == -1 || occupation[idx_prev] == 0) {
              int idx_cur = salle_index(chemin[pos]);
              if (idx_cur >= 0) occupation[idx_cur] = 0;
              f->position--;
              if (salle_index(chemin[f->position]) >= 0)
                occupation[salle_index(chemin[f->position])] = 1;
              // Si on revient à Sv, on change de chemin (apprentissage)
              if (chemin[f->position] == "Sv") {
                f->chemin = 1; // Passer au bon chemin
                f->position = 0;
                f->retour = false;
              }
            }
          }
        }
        else if (ch == 1) {
          // Chemin correct vers Sd
          if (pos < (int)chemin.size() - 1) {
            std::string next = chemin[pos + 1];
            int idx_next = salle_index(next);
            if (next == "Sd" || (idx_next >= 0 && occupation[idx_next] == 0)) {
              int idx_cur = salle_index(chemin[pos]);
              if (idx_cur >= 0) occupation[idx_cur] = 0;
              f->position++;
              if (chemin[f->position] != "Sd" && salle_index(chemin[f->position]) >= 0)
                occupation[salle_index(chemin[f->position])] = 1;
              if (chemin[f->position] == "Sd")
                f->arrivee = true;
            }
          }
        }
      }
      
      // Compter les fourmis arrivées
      arrived = 0;
      for (int i = 0; i < nb_fourmis; ++i) {
        if (fourmis[i]->arrivee) arrived++;
      }
    }
  }
};

# Processus de backtracking

1. Une fourmi atteint S3 (impasse identifiée)
2. Activation du mode retour (fourmi.retour = true)
3. Parcours inverse : S3 → S2 → S1 → Sv
4. Changement d'itinéraire à Sv (fourmi.chemin = 1)
5. Réinitialisation de l'indicateur de retour (fourmi.retour = false)
6. Continuation sur le bon chemin : Sv → S1 → S4 → Sd
